section .data
    ; Define variables
    num1 dq 10          ; First number (64-bit)
    num2 dq 20          ; Second number (64-bit)

section .text
    global _start

_start:
    ; Load values into registers
    mov rax, [num1]     ; Load num1 into rax
    mov rbx, [num2]     ; Load num2 into rbx

    ; Perform addition
    add rax, rbx        ; Add num1 and num2, result in rax

    ; Print result
    mov rdi, 1          ; File descriptor for STDOUT
    mov rsi, rax        ; Result value to print
    mov rdx, 8          ; Length of the number in bytes (64-bit)
    mov rax, 1          ; syscall number for write
    syscall             ; Invoke syscall

    ; Exit program
    mov rax, 60         ; syscall for exit
    xor rdi, rdi        ; exit status 0
    syscall             ; Invoke syscall
